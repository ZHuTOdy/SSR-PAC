#include<stdio.h>

int main()
{
	//如果购买的三种商品有一种商品价格大于1000
	//或三种商品总金额大于5000，折扣率为30%
	
	double price1 = 229.0;
	double price2 = 10240.55;
	double price3 = 2345.3;
	double zongjia = price1 + price2 + price3;
	double zhekou;
	if (price1 > 1000 || price2 > 1000 || price3 > 1000 || zongjia > 5000)
	{
		zhekou = 0.3;
	}
	else
	{
		zhekou = 1;
	}
	zongjia = zhekou * zongjia;
	printf("总价是%.3f\n", zongjia);
	
	int month;
	printf("请输入月份，我来判断天数：");
	scanf_s("%d", &month);		//输入月份
	switch (month)
	{
	case 1:
	case 3:
	case 5:
	case 7:
	case 8:
	case 10:
	case 12:
		printf("有31天！\n");
		break;
	case 4:
	case 6:
	case 9:
	case 11:
		printf("有30天！\n");
		break;
	case 2:
		printf("有28天！\n");
		break;
	}
	return 0;
}
